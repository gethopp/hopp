+#[derive(Debug, Clone, Copy, PartialEq, Eq)]
+pub enum SystemTheme {
+    Light,
+    Dark,
+}
+
+#[cfg(target_os = "macos")]
+pub mod macos {
+    use super::SystemTheme;
+    // Removed unused imports
+    use std::sync::mpsc;
+    use std::thread;
+    use std::time::Duration;
+
+    pub fn get_system_theme() -> SystemTheme {
+        // For now, let's use a simpler approach by checking the system defaults
+        // This is a simplified version that should work
+        use std::process::Command;
+
+        let output = Command::new("defaults")
+            .args(&["read", "-g", "AppleInterfaceStyle"])
+            .output();
+
+        match output {
+            Ok(output) => {
+                let stdout = String::from_utf8_lossy(&output.stdout);
+                if stdout.trim() == "Dark" {
+                    SystemTheme::Dark
+                } else {
+                    SystemTheme::Light
+                }
+            }
+            Err(_) => SystemTheme::Light, // Default to light if we can't determine
+        }
+    }
+
+    /// Start monitoring system theme changes
+    pub fn start_theme_monitoring<F>(callback: F) -> mpsc::Sender<()>
+    where
+        F: Fn(SystemTheme) + Send + 'static,
+    {
+        let (tx, rx) = mpsc::channel();
+
+        thread::spawn(move || {
+            let mut current_theme = get_system_theme();
+            callback(current_theme);
+
+            loop {
+                // Check if we should stop monitoring
+                if rx.try_recv().is_ok() {
+                    break;
+                }
+
+                thread::sleep(Duration::from_secs(1));
+
+                let new_theme = get_system_theme();
+                if new_theme != current_theme {
+                    current_theme = new_theme;
+                    callback(new_theme);
+                }
+            }
+        });
+
+        tx
+    }
+}
+
+#[cfg(not(target_os = "macos"))]
+pub mod fallback {
+    use super::SystemTheme;
+    use std::sync::mpsc;
+
+    pub fn get_system_theme() -> SystemTheme {
+        // Default to light theme on non-macOS platforms
+        SystemTheme::Light
+    }
+
+    pub fn start_theme_monitoring<F>(_callback: F) -> mpsc::Sender<()>
+    where
+        F: Fn(SystemTheme) + Send + 'static,
+    {
+        let (tx, _rx) = mpsc::channel();
+        // No-op for non-macOS platforms
+        tx
+    }
+}
+

+pub use macos::*;
+
+#[cfg(not(target_os = "macos"))]
+pub use fallback::*;
